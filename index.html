// ===================================================================== // SẢN LƯỢNG XẾP DỠ — Module báo cáo tuần/tháng/quý/năm // Import file Excel từ TOS → pivot → lưu sheet tổng hợp // ===================================================================== const SLXD_DAILY_SHEET = 'slxd_daily'; const SLXD_CUSTOMER_SHEET = 'slxd_customer'; const SLXD_IMPORT_LOG = 'slxd_import_log'; const SLXD_HEADERS = { slxd_daily: ['ngay','loai','phuong_an','phuong_an_full','kich_co','fe','so_luong','teus','trong_luong_tb','phuong_tien','grade_a','grade_b'], slxd_customer: ['thang','khach_hang','phuong_an','so_luong','teus'], slxd_import_log: ['seq','loai','tu_ngay','den_ngay','file_name','so_dong_raw','so_dong_daily','so_dong_customer','imported_at','imported_by'] }; // ── SETUP ── function setupSlxdModule() { var ss = SpreadsheetApp.openById(SPREADSHEET_ID); var results = []; for (var name in SLXD_HEADERS) { var headers = SLXD_HEADERS[name]; var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight('bold') .setBackground('#0f3460').setFontColor('#00d4ff'); sheet.setFrozenRows(1); results.push(name + ': created'); } else { // Bổ sung cột thiếu var lastCol = Math.max(sheet.getLastColumn() || 1, 1); var existing = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(function(v) { return String(v).trim(); }); var added = 0; headers.forEach(function(h) { if (existing.indexOf(h) < 0) { sheet.getRange(1, existing.length + 1).setValue(h).setFontWeight('bold') .setBackground('#0f3460').setFontColor('#00d4ff'); existing.push(h); added++; } }); results.push(name + ': exists' + (added ? ' (+' + added + ')' : '')); } } return { success: true, results: results }; } // ===================================================================== // IMPORT — Nhận raw data từ frontend (đã parse bằng SheetJS), pivot, lưu // ===================================================================== function importSlxd(data) { // data.loai = 'tuan' | 'thang' // data.tu_ngay = '2025-12-01' // data.den_ngay = '2025-12-07' // data.rows = array of objects (raw data từ Excel, đã parse ở frontend) // data.file_name = tên file if (!data.rows || !data.rows.length) return { success: false, error: 'Không có dữ liệu' }; if (!data.loai) return { success: false, error: 'Thiếu loại (tuan/thang)' }; if (!data.tu_ngay || !data.den_ngay) return { success: false, error: 'Thiếu khoảng ngày' }; var rawRows = data.rows; var loai = data.loai; var tuNgay = data.tu_ngay; var denNgay = data.den_ngay; // ── Pivot daily ── var dailyMap = {}; // key: ngay|phuong_an|kich_co|fe|phuong_tien var customerMap = {}; // key: thang|khach_hang|phuong_an rawRows.forEach(function(r) { // Xác định phương án tác nghiệp (dùng cột đầy đủ) var phuongAnFull = String(r['Phương án tác nghiệp'] || '').trim(); // Lấy mã viết tắt từ phương án tác nghiệp (VD: "NRCK: Nhập rỗng cảng khác" → "NRCK") var phuongAn = phuongAnFull.split(':')[0].trim() || String(r['Phương Án'] || '').trim(); // Nhập/Hạ → dùng "Cont vào bãi", Cấp/Xuất/Chuyển → dùng "Cont ra bãi" var nhapGroup = ['NRCK', 'NRKH', 'HRKH', 'HBCX']; var isNhap = nhapGroup.indexOf(phuongAn) >= 0; var ngaySrc = isNhap ? r['Cont vào bãi'] : r['Cont ra bãi']; var ngay = parseSlxdDate_(ngaySrc); if (!ngay) return; var ngayStr = fmtSlxdDate_(ngay); var thangStr = fmtSlxdMonth_(ngay); var kichCo = String(r['Kích cỡ'] || '').trim(); var fe = String(r['F/E'] || '').trim(); var trongLuong = parseFloat(r['Trọng lượng']) || 0; var phuongTien = String(r['Phương tiện'] || '').trim(); var grade = String(r['Container Grade'] || '').trim(); var khachHang = String(r['Chủ Hàng'] || '').trim(); // Tính TEUs: 20ft=1, 40ft=2 var teus = kichCo.indexOf('20') >= 0 ? 1 : 2; var isGradeA = grade.indexOf('A') >= 0 ? 1 : 0; var isGradeB = grade.indexOf('B') >= 0 ? 1 : 0; // Daily pivot — key dùng mã phương án tác nghiệp đầy đủ var dk = ngayStr + '|' + phuongAn + '|' + kichCo + '|' + fe + '|' + phuongTien; if (!dailyMap[dk]) dailyMap[dk] = { sl: 0, teus: 0, tl: 0, ga: 0, gb: 0, paf: phuongAnFull }; dailyMap[dk].sl++; dailyMap[dk].teus += teus; dailyMap[dk].tl += trongLuong; dailyMap[dk].ga += isGradeA; dailyMap[dk].gb += isGradeB; // Customer pivot if (khachHang) { var khClean = khachHang.replace(/^\d+\s*:\s*/, '').trim(); var ck = thangStr + '|' + khClean + '|' + phuongAn; if (!customerMap[ck]) customerMap[ck] = { sl: 0, teus: 0 }; customerMap[ck].sl++; customerMap[ck].teus += teus; } }); // ── Xoá data cũ trong khoảng ngày ── deleteSlxdRange_(SLXD_DAILY_SHEET, SLXD_HEADERS.slxd_daily, tuNgay, denNgay, loai); // Xoá customer data theo tháng (lấy tất cả tháng trong khoảng) var months = getMonthsInRange_(tuNgay, denNgay); months.forEach(function(m) { deleteSlxdCustomerMonth_(m); }); // ── Ghi daily ── var dailySheet = getOrCreateSheet_(SLXD_DAILY_SHEET, SLXD_HEADERS.slxd_daily); var dailyRows = []; for (var dk in dailyMap) { var parts = dk.split('|'); var d = dailyMap[dk]; dailyRows.push([ parts[0], // ngay loai, // loai parts[1], // phuong_an d.paf, // phuong_an_full parts[2], // kich_co parts[3], // fe d.sl, // so_luong d.teus, // teus d.sl > 0 ? Math.round(d.tl / d.sl * 10) / 10 : 0, // trong_luong_tb parts[4], // phuong_tien d.ga, // grade_a d.gb // grade_b ]); } if (dailyRows.length) { dailySheet.getRange(dailySheet.getLastRow() + 1, 1, dailyRows.length, SLXD_HEADERS.slxd_daily.length).setValues(dailyRows); } // ── Ghi customer ── var custSheet = getOrCreateSheet_(SLXD_CUSTOMER_SHEET, SLXD_HEADERS.slxd_customer); var custRows = []; for (var ck in customerMap) { var parts = ck.split('|'); var c = customerMap[ck]; custRows.push([parts[0], parts[1], parts[2], c.sl, c.teus]); } if (custRows.length) { custSheet.getRange(custSheet.getLastRow() + 1, 1, custRows.length, SLXD_HEADERS.slxd_customer.length).setValues(custRows); } // ── Lưu log ── var logSheet = getOrCreateSheet_(SLXD_IMPORT_LOG, SLXD_HEADERS.slxd_import_log); var now = Utilities.formatDate(new Date(), 'Asia/Ho_Chi_Minh', 'dd/MM/yyyy HH:mm'); logSheet.appendRow([ nextSeq(logSheet), loai, tuNgay, denNgay, data.file_name || '', rawRows.length, dailyRows.length, custRows.length, now, data.imported_by || '' ]); return { success: true, raw_rows: rawRows.length, daily_rows: dailyRows.length, customer_rows: custRows.length, months: months }; } // ── Xoá data daily theo khoảng ngày + loại ── function deleteSlxdRange_(sheetName, headers, tuNgay, denNgay, loai) { var sheet; try { sheet = getOrCreateSheet_(sheetName, headers); } catch(e) { return; } var d = sheet.getDataRange().getValues(); if (d.length < 2) return; var from = parseDdMmYyyy_(tuNgay), to = parseDdMmYyyy_(denNgay); if (!from || !to) return; for (var i = d.length - 1; i >= 1; i--) { var rowLoai = String(d[i][1]).trim(); if (rowLoai !== loai) continue; var rowDate = parseDdMmYyyy_(String(d[i][0]).trim()); if (rowDate && rowDate >= from && rowDate <= to) { sheet.deleteRow(i + 1); } } } function parseDdMmYyyy_(s) { if (!s) return null; var p = String(s).split('/'); if (p.length < 3) return null; return new Date(+p[2], +p[1]-1, +p[0]); } // ── Xoá customer data theo tháng ── function deleteSlxdCustomerMonth_(thang) { var sheet; try { sheet = getOrCreateSheet_(SLXD_CUSTOMER_SHEET, SLXD_HEADERS.slxd_customer); } catch(e) { return; } var d = sheet.getDataRange().getValues(); for (var i = d.length - 1; i >= 1; i--) { if (String(d[i][0]).trim() === thang) sheet.deleteRow(i + 1); } } // ── Lấy danh sách tháng trong khoảng ── function getMonthsInRange_(tuNgay, denNgay) { // tuNgay, denNgay format: dd/MM/yyyy var months = {}; var dp1 = tuNgay.split('/'), dp2 = denNgay.split('/'); var d1 = new Date(+dp1[2], +dp1[1]-1, +dp1[0]); var d2 = new Date(+dp2[2], +dp2[1]-1, +dp2[0]); var cur = new Date(d1.getFullYear(), d1.getMonth(), 1); while (cur <= d2) { var m = String(cur.getMonth()+1).padStart(2,'0') + '/' + cur.getFullYear(); months[m] = true; cur.setMonth(cur.getMonth() + 1); } return Object.keys(months); } // ===================================================================== // API — Đọc data tổng hợp cho frontend // ===================================================================== function getSlxdDaily(params) { var sheet = getOrCreateSheet_(SLXD_DAILY_SHEET, SLXD_HEADERS.slxd_daily); var d = sheet.getDataRange().getValues(); if (d.length < 2) return { success: true, data: [] }; var h = d[0]; var data = []; for (var i = 1; i < d.length; i++) { var row = {}; for (var j = 0; j < h.length; j++) row[h[j]] = (d[i][j] !== null && d[i][j] !== undefined) ? d[i][j] : ''; data.push(row); } return { success: true, data: data }; } function getSlxdCustomer() { var sheet = getOrCreateSheet_(SLXD_CUSTOMER_SHEET, SLXD_HEADERS.slxd_customer); var d = sheet.getDataRange().getValues(); if (d.length < 2) return { success: true, data: [] }; var h = d[0]; var data = []; for (var i = 1; i < d.length; i++) { var row = {}; for (var j = 0; j < h.length; j++) row[h[j]] = (d[i][j] !== null && d[i][j] !== undefined) ? d[i][j] : ''; data.push(row); } return { success: true, data: data }; } function getSlxdImportLog() { var sheet = getOrCreateSheet_(SLXD_IMPORT_LOG, SLXD_HEADERS.slxd_import_log); var d = sheet.getDataRange().getValues(); if (d.length < 2) return { success: true, data: [] }; var h = d[0]; var data = []; for (var i = 1; i < d.length; i++) { var row = {}; for (var j = 0; j < h.length; j++) row[h[j]] = (d[i][j] !== null && d[i][j] !== undefined) ? d[i][j] : ''; data.push(row); } data.reverse(); return { success: true, data: data }; } // ── Helpers ── function parseSlxdDate_(v) { if (!v) return null; // Excel serial number var n = parseFloat(v); if (!isNaN(n) && n > 40000 && n < 60000) { return new Date((n - 25569) * 86400000); } // String date var d = new Date(v); return isNaN(d) ? null : d; } function fmtSlxdDate_(d) { if (!d) return ''; return String(d.getDate()).padStart(2,'0') + '/' + String(d.getMonth()+1).padStart(2,'0') + '/' + d.getFullYear(); } function fmtSlxdMonth_(d) { if (!d) return ''; return String(d.getMonth()+1).padStart(2,'0') + '/' + d.getFullYear(); }
